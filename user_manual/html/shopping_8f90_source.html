49  SUBROUTINE hopping(my_rho,v,r,trajlabel)
50 
51  IMPLICIT NONE
52 
53  INTEGER ,INTENT(IN) :: trajlabel
54  COMPLEX(KIND=DP),INTENT(IN) :: my_rho(nstates,nstates)
55  REAL(KIND=dp) ,INTENT(INOUT) :: v(n_dof)
56  REAL(KIND=dp) ,INTENT(IN) :: r(n_dof)
57  INTEGER :: i_state,i_dof
58  REAL(KIND=dp) :: scal2,re_rhoij,rhojj, &
59  hop_prob(nstates),deltae,&
60  dist,xi
61 
62  hop_prob = 0.0_dp
63 
64  DO i_state=1,nstates
65  IF(i_state/=occ_state(trajlabel)) THEN
66  IF(typ_cal=="TSHFS") THEN
67  scal2 = 0.0_dp
68  DO i_dof=1,n_dof
69  scal2 = scal2 + coup(trajlabel,occ_state(trajlabel),i_state,i_dof) * v(i_dof)
70  END DO
71  re_rhoij = real(my_rho(occ_state(trajlabel),i_state) ,kind=dp)
72  rhojj = real(my_rho(occ_state(trajlabel),occ_state(trajlabel)),kind=dp)
73  hop_prob(i_state) = 2.0_dp * dt * scal2 * re_rhoij/rhojj
74  ENDIF
75  IF(typ_cal=="TSHLZ") THEN
76  CALL xi_for_model_system(dist,deltae,xi,r(1),v(1),trajlabel)
77  IF(lz_hop(trajlabel)==0) THEN
78  IF(.NOT. new_potential) THEN
79  WRITE(*,*) "Landau-Zener not yet available"
80  stop
81  ENDIF
82  IF(deltae<adia_nrg_gap .AND. dist<lz_dist_cutoff) THEN
83  hop_prob(i_state) = exp(-2._dp*pi*xi)
84  count_traj=count_traj+1
85  lz_hop(trajlabel)=lz_hop(trajlabel)+1
86  !WRITE(*,* ) occ_state(trajlabel), "to", i_state, 'with prob', hop_prob(i_state)
87  ENDIF
88  ENDIF
89  ENDIF
90  ENDIF
91  END DO
92 
93  IF(count_traj/=0) WRITE(*,*) count_traj
94  IF(typ_cal=="TSHLZ" .AND. deltae>adia_nrg_gap .AND. &
95  dist>lz_dist_cutoff .AND. count_traj==ntraj) THEN
96  lz_hop(trajlabel)=0
97  IF(sum(lz_hop)==0) count_traj=0
98  ENDIF
99  !IF(typ_cal=="TSHLZ") LZ_deltaE(2,trajlabel)=LZ_deltaE(1,trajlabel)
100 
101  CALL choose_bostate(v,hop_prob,trajlabel)
102 
103  END SUBROUTINE hopping

120  SUBROUTINE choose_bostate(v,hop_prob,trajlabel)
121 
122  IMPLICIT NONE
123 
124  INTEGER ,INTENT(IN) :: trajlabel
125  REAL(KIND=dp),INTENT(INOUT) :: v(n_dof),hop_prob(nstates)
126  REAL(KIND=dp) :: myrand(1),prob_sum(nstates)
127  INTEGER :: i_state,j_state,old_occ_state
128 
129  old_occ_state = occ_state(trajlabel)
130 
131  CALL random_number(myrand)
132 
133  prob_sum = 0.0_dp
134 
135  DO i_state=1,nstates
136  DO j_state=1,i_state
137  prob_sum(i_state) = prob_sum(i_state) + hop_prob(j_state)
138  END DO
139  END DO
140 
141  stateloop: DO i_state=1,nstates
142 
143  IF(i_state/=old_occ_state) THEN
144 
145  IF(i_state==1) THEN
146  IF(myrand(1)>0.0_dp .AND. myrand(1)<=prob_sum(i_state)) THEN
147  occ_state(trajlabel) = i_state
148  EXIT stateloop
149  END IF
150  ELSE
151  IF(myrand(1)>prob_sum(i_state-1) .AND. myrand(1)<=prob_sum(i_state)) THEN
152  occ_state(trajlabel) = i_state
153  EXIT stateloop
154  END IF
155  END IF
156 
157  END IF
158 
159  END DO stateloop
160 
161  IF(occ_state(trajlabel)/=old_occ_state) THEN
162  CALL momentum_correction(v,old_occ_state,trajlabel)
163  END IF
164 
165  END SUBROUTINE choose_bostate

186  SUBROUTINE momentum_correction(v,old_occ_state,trajlabel)
187 
188  IMPLICIT NONE
189 
190  REAL(KIND=dp),INTENT(INOUT) :: v(n_dof)
191  INTEGER ,INTENT(IN) :: old_occ_state,trajlabel
192  REAL(KIND=dp) :: deltae,scal1,scal2, &
193  energy_check,scaling_factor
194  INTEGER :: i_dof
195 
196  scal1 = 0.0_dp
197  scal2 = 0.0_dp
198 
199  deltae = boenergy(trajlabel,old_occ_state) - boenergy(trajlabel,occ_state(trajlabel))
200 
201  DO i_dof=1,n_dof
202  scal1 = scal1 + coup(trajlabel,old_occ_state,occ_state(trajlabel),i_dof)**2 / mass(i_dof) !d^v(kl)**2/M^v
203  scal2 = scal2 + v(i_dof) * coup(trajlabel,old_occ_state,occ_state(trajlabel),i_dof) !V * d(kl)
204  END DO
205 
206  energy_check = scal2**2 + 2.0_dp * scal1 * deltae
207 
208  IF(energy_check >= 0.0_dp) THEN
209 
210  IF(scal2 >= 0.0_dp) THEN
211  scaling_factor = ( scal2 - sqrt(energy_check) )/ scal1
212  ELSE
213  scaling_factor = ( scal2 + sqrt(energy_check) )/ scal1
214  END IF
215  v(:) = v(:) - scaling_factor * coup(trajlabel,old_occ_state,occ_state(trajlabel),:) / mass(:)
216  ELSE
217  occ_state(trajlabel) = old_occ_state
218  END IF
219 
220  END SUBROUTINE momentum_correction

239  SUBROUTINE decoherence_coorection(coeff,v,trajlabel)
240 
241  IMPLICIT NONE
242 
243  COMPLEX(KIND=QP),INTENT(INOUT) :: coeff(nstates)
244  REAL(KIND=dp) ,INTENT(IN) :: v(n_dof)
245  real(kind=dp) :: decay_time,deltae, &
246  kinetic_energy, &
247  sum_rho
248  INTEGER :: i_dof,i_state, &
249  trajlabel
250 
251  kinetic_energy = 0.0_dp
252  sum_rho = 0.0_dp
253 
254  DO i_dof=1,n_dof
255  kinetic_energy = kinetic_energy + (mass(i_dof) * v(i_dof)**2) / 2_dp
256  END DO
257 
258  DO i_state=1,nstates
259  IF(i_state/=occ_state(trajlabel)) THEN
260  deltae = boenergy(trajlabel,i_state) - boenergy(trajlabel,occ_state(trajlabel))
261  decay_time = ( 1_dp + c_parameter / kinetic_energy) / abs(deltae)
262  coeff(i_state) = coeff(i_state) * exp(-dt / decay_time)
263  sum_rho = sum_rho + real(conjg(coeff(i_state)) * coeff(i_state))
264  END IF
265  END DO
266  coeff(occ_state(trajlabel)) = coeff(occ_state(trajlabel)) * &
267  sqrt( ( 1_dp - sum_rho ) / ( conjg(coeff(occ_state(trajlabel))) * coeff(occ_state(trajlabel)) ) )
268  END SUBROUTINE decoherence_coorection
