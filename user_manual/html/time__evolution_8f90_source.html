54

67  ! The random number generator for the surface hopping algorithm is initialized
68 
69  timeloop: DO time=1,nsteps
70 
71  trajsloop: DO itraj=1,ntraj
72 
73  IF (typ_cal=="TSHLZ" .OR. typ_cal=="TSHFS") THEN
74  IF(type_deco=="CT") &
75  CALL accumulated_boforce(bocoeff(itraj,:),my_force(itraj,:,:),itraj)
76  ! The accumulated force is compuetd
77  CALL update_velocity(vcl(itraj,:),boforce(itraj,occ_state(itraj),:))
78  ! Velocities are updated at half time step
79  CALL update_position(rcl(itraj,:),vcl(itraj,:))
80  ! Positions are updated at a full time step
81  CALL boproblem(rcl(itraj,:),itraj)
82  ! Electronic properties are computed at the new positions
83  CALL update_velocity(vcl(itraj,:),boforce(itraj,occ_state(itraj),:))
84  ! Velocities are updated at half time step
85  CALL rk4_coeff(vcl(itraj,:),bocoeff(itraj,:),k_li(itraj,:,:),itraj)
86  ! Electronic coefficients are updated
87  IF (type_deco=="ED") &
88  CALL decoherence_coorection(bocoeff(itraj,:),vcl(itraj,:),itraj)
89  ! Electronic coefficients are corrected based on decoherence corrections
90  CALL hopping(bosigma(itraj,:,:),vcl(itraj,:),rcl(itraj,:),itraj)
91  ! Count hoppings for each trajectory
92  END IF
93 
94  IF (typ_cal=="CTMQC") THEN
95  CALL accumulated_boforce(bocoeff(itraj,:),my_force(itraj,:,:),itraj)
96  ! The accumulated force is compuetd
97  IF(f_correction) THEN
98  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
99  my_force_e(itraj,:,:),k_li(itraj,:,:),itraj)
100  ELSE
101  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
102  my_force(itraj,:,:),k_li(itraj,:,:),itraj)
103  ENDIF
104  ! The classical force is computed
105  CALL update_velocity(vcl(itraj,:),classical_force)
106  ! Velocities are updated at half time step
107  CALL update_position(rcl(itraj,:),vcl(itraj,:))
108  ! Positions are updated at a full time step
109  CALL boproblem(rcl(itraj,:),itraj)
110  ! Electronic properties are computed at the new positions
111  IF(f_correction) THEN
112  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
113  my_force_e(itraj,:,:),k_li(itraj,:,:),itraj)
114  ELSE
115  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
116  my_force(itraj,:,:),k_li(itraj,:,:),itraj)
117  ENDIF
118  ! The classical force is computed
119  CALL update_velocity(vcl(itraj,:),classical_force)
120  ! Velocities are updated at half time step
121  CALL rk4_coeff(vcl(itraj,:),bocoeff(itraj,:),k_li(itraj,:,:),itraj)
122  ! Electronic coefficients are updated
123  END IF
124 
125  IF (typ_cal=="EHREN") THEN
126  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
127  my_force(itraj,:,:),k_li(itraj,:,:),itraj)
128  ! The classical force is computed
129  CALL update_velocity(vcl(itraj,:),classical_force)
130  ! Velocities are updated at half time step
131  CALL update_position(rcl(itraj,:),vcl(itraj,:))
132  ! Positions are updated at a full time step
133  CALL boproblem(rcl(itraj,:),itraj)
134  ! Electronic properties are computed at the new positions
135  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
136  my_force(itraj,:,:),k_li(itraj,:,:),itraj)
137  ! The classical force is computed
138  CALL update_velocity(vcl(itraj,:),classical_force)
139  ! Velocities are updated at half time step
140  CALL rk4_coeff(vcl(itraj,:),bocoeff(itraj,:),k_li(itraj,:,:),itraj)
141  ! Electronic coefficients are updated
142  END IF
143 
144  END DO trajsloop
145 
146  DO i=1,nstates
147  DO j=1,nstates
148  bosigma(:,i,j) = conjg(bocoeff(:,i)) * bocoeff(:,j)
149  END DO
150  END DO
151 
152  IF(mod(time,dump)==0) CALL plot(bosigma,rcl,vcl,time)
153 
154  IF(typ_cal=="CTMQC" .OR. type_deco=="CT") THEN
155  IF(f_correction) THEN
156  !CTMQC-E
157  CALL acc_force_ec(vcl,bocoeff(:,:),my_force,my_force_e)
158  CALL quantum_momentum(rcl,my_force_e,bosigma,k_li)
159  IF(mod(time,dump)==0) CALL plot_stc(k_li,bosigma,my_force_e,vcl,time)
160  ELSE
161  !CTMQC
162  CALL quantum_momentum(rcl,my_force,bosigma,k_li)
163  IF(mod(time,dump)==0) CALL plot_stc(k_li,bosigma,my_force,vcl,time)
164  ENDIF
165  ENDIF
166  END DO timeloop
167 
168  CALL finalize_local_vars
169

206  SUBROUTINE initialize_local_vars
207 
208  INTEGER :: itraj,i,j
209 
210  ALLOCATE(rcl(ntraj,n_dof),vcl(ntraj,n_dof), &

235  ENDIF
236  IF(type_deco=="CT") THEN
237  DO i=1,n_init_bo
238  bocoeff(:,initial_bostate(i)) = &
239  cmplx(sqrt(weight_bostate(i)),0.0_dp,qp)
240  ENDDO
241  ENDIF
242  ENDIF
243 
244  DO itraj=1,ntraj
245  DO i=1,nstates
246  DO j=1,nstates
247  bosigma(itraj,i,j) = conjg(bocoeff(itraj,i)) * bocoeff(itraj,j)
248  END DO
249  END DO
250  rcl(itraj,:) = initial_positions(itraj,:)
251  CALL boproblem(rcl(itraj,:),itraj)
252  CALL non_adiabatic_force(bocoeff(itraj,:),classical_force, &
253  my_force(itraj,:,:),k_li(itraj,:,:),itraj)
254  vcl(itraj,:) = initial_momenta(itraj,:) / mass
255  END DO
256 
257  END SUBROUTINE initialize_local_vars
