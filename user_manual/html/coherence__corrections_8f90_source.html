50  SUBROUTINE accumulated_boforce(coeff,force,trajlabel)
51 
52  INTEGER, INTENT(IN) :: trajlabel
53  COMPLEX(KIND=QP),INTENT(IN) :: coeff(nstates)
54  REAL(KIND=dp), INTENT(INOUT) :: force(n_dof,nstates)
55  INTEGER :: i,i_dof,check
56  REAL(KIND=dp), PARAMETER :: threshold = 0.005_dp
57  REAL(KIND=dp), ALLOCATABLE :: mean_force(:)
58 
59  ALLOCATE(mean_force(n_dof),stat=check)
60  IF(check/=0) print*,'error 1 mean_force in accumulated_BOforce'
61 
62  IF(list_coupled_trajectories(trajlabel)==1) THEN
63 
64  mean_force=0.0_dp
65  DO i_dof=1,n_dof
66  DO i=1,nstates
67  mean_force(i_dof)=mean_force(i_dof)+&
68  abs(coeff(i))**2*boforce(trajlabel,i,i_dof)
69  ENDDO
70  ENDDo
71 
72  DO i=1,nstates
73  IF(abs(coeff(i))**2>threshold .AND. abs(coeff(i))**2<1.0_dp-threshold) THEN
74  DO i_dof=1,n_dof
75  IF(abs(boforce(trajlabel,i,i_dof)-mean_force(i_dof))<=zero) THEN
76  force(i_dof,i)=0.0_dp
77  ELSE
78  force(i_dof,i)=force(i_dof,i)+dt*(boforce(trajlabel,i,i_dof)-mean_force(i_dof))
79  END If
80  END Do
81  ELSE
82  force(:,i)=0.0_dp
83  END IF
84  END DO
85 
86  ENDIF
87 
88  DEALLOCATE(mean_force,stat=check)
89  IF(check/=0) print*,'error 2 mean_force in accumulated_BOforce'
90 
91  IF(ntraj==1) force=0.0_dp
92 
93  END SUBROUTINE accumulated_boforce

132  SUBROUTINE quantum_momentum(Rcl,acc_force,BOsigma,k_li)
133 
134  USE output
135 
136  COMPLEX(KIND=QP),INTENT(IN) :: BOsigma(ntraj,nstates,nstates)
137  REAL(KIND=dp),INTENT(IN) :: rcl(ntraj,n_dof), &
138  acc_force(ntraj,n_dof,nstates)
139  REAL(KIND=dp),INTENT(INOUT) :: k_li(ntraj,nstates,nstates)
140  INTEGER :: itraj,jtraj,i_dof,index_ij, &
141  istate,jstate
142  REAL(KIND=dp),ALLOCATABLE :: gamma(:,:),g_i(:),prod_g_i(:,:), &
143  w_ij(:,:,:),slope_i(:,:), &
144  ratio(:,:,:),num_old(:,:,:), &
145  num_new(:,:,:),num(:,:,:), &
146  denom(:,:),qmom(:,:,:), &
147  threshold(:)
148  REAL(KIND=dp) :: distance
149 
150  ALLOCATE(gamma(n_dof,ntraj),g_i(ntraj),prod_g_i(ntraj,ntraj), &
151  w_ij(n_dof,ntraj,ntraj),slope_i(n_dof,ntraj), &
152  ratio(n_dof,npairs,ntraj),num_old(n_dof,ntraj,npairs), &
153  num_new(n_dof,ntraj,npairs),num(n_dof,ntraj,npairs), &
154  denom(n_dof,npairs),qmom(n_dof,ntraj,npairs), &
155  threshold(n_dof))
156 
157  gamma = 0.0_dp
158 
159  IF(model_potential=="tully") THEN
160  threshold(:)=0.1d0*sigma(:)
161  ELSE
162  threshold(:) = sigma(:)
163  ENDIF
164  DO i_dof=1,n_dof
165  IF(sigma(i_dof)==0.0_dp) THEN
166  threshold(i_dof)=1.0_dp/var_momentum(i_dof)
167  ENDIF
168  ENDDO
169 
170  DO i_dof=1,n_dof
171  gamma(i_dof,:)=sigma(i_dof)
172  IF(gamma(i_dof,1)==0.0_dp) gamma(i_dof,:)=1._dp/var_momentum(i_dof)
173  IF(model_potential=="tully") gamma(i_dof,:)=0.10_dp*sigma(i_dof)
174  ENDDO
175 
176  DO itraj=1,ntraj
177  g_i(itraj)=0.0_dp
178  !This part computes the nuclear density as the sum of normalized
179  !Gaussians centered at the positions of the trajectories (labeled jtraj)
180  !with variances as computed above
181  !!!!! I need to know the value of the nuclear density at the position
182  !!!!! of the trajectory (itraj)
183  DO jtraj=1,ntraj
184  prod_g_i(itraj,jtraj) = 1.0_dp
185  DO i_dof=1,n_dof
186  prod_g_i(itraj,jtraj)=prod_g_i(itraj,jtraj)* &
187  (dexp(-(rcl(itraj,i_dof)-rcl(jtraj,i_dof))**2/ &
188  (gamma(i_dof,jtraj))**2/2._dp))* &
189  (1.0_dp/sqrt(2.0_dp*pi*(gamma(i_dof,jtraj))**2))
190  END Do
191  g_i(itraj) = g_i(itraj) + prod_g_i(itraj,jtraj)
192  END Do
193  END DO
194 
195  !W_ij => see SI of paper Min, Agostini, Tavernelli, Gross for its definition
196  !This part computes W_ij, whose sum over j (jtraj in the loop below) is the
197  !slope of the quantum momentum when a sum of Gaussians is used
198  !to approximate the nuclear density
199  DO itraj=1,ntraj
200  DO jtraj=1,ntraj
201  DO i_dof=1,n_dof
202  w_ij(i_dof,itraj,jtraj)=prod_g_i(itraj,jtraj)/ &
203  2.0_dp/(gamma(i_dof,jtraj))**2/g_i(itraj)
204  END DO
205  END Do
206  END DO
207  !The slope is calculated here as a sum over j of W_ij
208  slope_i=0.0_dp
209  DO itraj=1,ntraj
210  DO i_dof=1,n_dof
211  DO jtraj=1,ntraj
212  slope_i(i_dof,itraj)=slope_i(i_dof,itraj)- &
213  w_ij(i_dof,itraj,jtraj)
214  END DO
215  END DO
216  END DO
217 
218  !Here I compute the center of the quantum momentum
219  !See SI of paper Min, Agostini, Tavernelli, Gross for the expression Eq.(28)
220  DO i_dof=1,n_dof
221  index_ij=0
222  DO istate=1,nstates
223  DO jstate=istate+1,nstates
224  index_ij=index_ij+1
225  ! denominator
226  denom(i_dof,index_ij)=0._dp
227  DO jtraj=1,ntraj
228  denom(i_dof,index_ij)=denom(i_dof,index_ij)+ &
229  real(BOsigma(jtraj,istate,istate),kind=dp)* &
230  real(BOsigma(jtraj,jstate,jstate),kind=dp)* &
231  (acc_force(jtraj,i_dof,istate)- &
232  acc_force(jtraj,i_dof,jstate))*slope_i(i_dof,jtraj)
233  END DO
234  END DO
235  END DO
236  END Do
237 
238  ratio=0.0_dp
239  DO itraj=1,ntraj
240  DO i_dof=1,n_dof
241  index_ij=0
242  DO istate=1,nstates
243  DO jstate=istate+1,nstates
244  index_ij=index_ij+1
245  ! numerator
246  num_old(i_dof,itraj,index_ij)= &
247  REAL(BOsigma(itraj,istate,istate),KIND=dp)* &
248  REAL(BOsigma(itraj,jstate,jstate),KIND=dp)* &
249  (acc_force(itraj,i_dof,istate)-acc_force(itraj,i_dof,jstate))* &
250  rcl(itraj,i_dof)*slope_i(i_dof,itraj) !pos
251  ! ratio
252  IF(abs(denom(i_dof,index_ij)) .LT. 0.00000001_dp) THEN
253  ratio(i_dof,index_ij,itraj)=0._dp
254  ELSE
255  ratio(i_dof,index_ij,itraj)=num_old(i_dof,itraj,index_ij)/&
256  denom(i_dof,index_ij) !pos
257  END If
258  END DO
259  END DO
260  END DO
261  END DO
262 
263  !Here I acutally compute the sum over I of Eq.(28) of SI of
264  !paper Min, Agostini, Tavernelli, Gross
265  DO itraj=1,ntraj
266  DO i_dof=1,n_dof
267  index_ij=0
268  DO istate=1,nstates
269  DO jstate=istate+1,nstates
270  index_ij=index_ij+1
271  num_old(i_dof,itraj,index_ij)=0._dp
272  DO jtraj=1,ntraj
273  num_old(i_dof,itraj,index_ij)=num_old(i_dof,itraj,index_ij)+ &!pos
274  ratio(i_dof,index_ij,jtraj)
275  END DO
276  IF(abs(slope_i(i_dof,itraj)) .LT. 0.0000001_dp .OR. &
277  num_old(i_dof,itraj,index_ij) .EQ. 0.0_dp) THEN
278  num_old(i_dof,itraj,index_ij)=rcl(itraj,i_dof)
279  END IF
280  END DO
281  END DO
282  END DO
283  END DO
284 
285  !Here the centers of the quantum momentum are cmoputed from the expression (21)
286  !of SI of paper Min, Agostini, Tavernelli, Gross. This is just to be sure that
287  !if the previous
288  DO itraj=1,ntraj
289  DO i_dof=1,n_dof
290  index_ij=0
291  DO istate=1,nstates
292  DO jstate=istate+1,nstates
293  index_ij=index_ij+1
294  num_new(i_dof,itraj,index_ij)=0.0_dp
295  IF(abs(slope_i(i_dof,itraj))<0.00000001_dp) THEN
296  num_new(i_dof,itraj,index_ij)=rcl(itraj,i_dof)
297  ELSE
298  DO jtraj=1,ntraj
299  num_new(i_dof,itraj,index_ij)=num_new(i_dof,itraj,index_ij)+ & !pos
300  rcl(jtraj,i_dof)*prod_g_i(itraj,jtraj)/2.0_dp/(gamma(i_dof,jtraj))**2/ &
301  g_i(itraj)/(-slope_i(i_dof,itraj))
302  END DO
303  END IF
304  END DO
305  END DO
306  END DO
307  END DO
308 
309  DO itraj=1,ntraj
310  DO i_dof=1,n_dof
311  index_ij=0
312  DO istate=1,nstates
313  DO jstate=istate+1,nstates
314  index_ij=index_ij+1
315  distance=dabs(num_old(i_dof,itraj,index_ij)-rcl(itraj,i_dof))
316  IF(distance>m_parameter(i_dof)*threshold(i_dof)) THEN
317  distance=dabs(num_new(i_dof,itraj,index_ij)-rcl(itraj,i_dof))
318  IF(distance>m_parameter(i_dof)*threshold(i_dof)) THEN
319  num(i_dof,itraj,index_ij)=rcl(itraj,i_dof)
320  ELSE
321  num(i_dof,itraj,index_ij)=num_new(i_dof,itraj,index_ij)
322  END IF
323  ELSE
324  num(i_dof,itraj,index_ij)=num_old(i_dof,itraj,index_ij)
325  END IF
326  END DO
327  END DO
328  END DO
329  END DO
330 
331  ! quantum momentum
332  DO itraj=1,ntraj
333  DO i_dof=1,n_dof
334  index_ij=0
335  DO istate=1,nstates
336  DO jstate=istate+1,nstates
337  index_ij=index_ij+1
338  qmom(i_dof,itraj,index_ij)=slope_i(i_dof,itraj)*(rcl(itraj,i_dof)-num(i_dof,itraj,index_ij))
339  END DO
340  END DO
341  END DO
342  END DO
343 
344  index_ij=0
345  DO istate=1,nstates
346  DO jstate=istate+1,nstates
347  index_ij=index_ij+1
348  DO itraj=1,ntraj
349  k_li(itraj,istate,jstate)=0.0_dp
350  k_li(itraj,jstate,istate)=0.0_dp
351  DO i_dof=1,n_dof
352  k_li(itraj,istate,jstate)=k_li(itraj,istate,jstate)+2.0_dp/mass(i_dof)* &
353  qmom(i_dof,itraj,index_ij)*acc_force(itraj,i_dof,istate)
354  k_li(itraj,jstate,istate)=k_li(itraj,jstate,istate)+2.0_dp/mass(i_dof)* &
355  qmom(i_dof,itraj,index_ij)*acc_force(itraj,i_dof,jstate)
356  END DO
357  END Do
358  END Do
359  END DO
360 
361  IF(ntraj==1) k_li=0.0_dp
362 
363  DEALLOCATE(gamma,g_i,prod_g_i,w_ij,slope_i, &
364  ratio,num_old,num_new,num,denom, &
365  qmom,threshold)
366 
367  END SUBROUTINE quantum_momentum

383  SUBROUTINE acc_force_ec(Vcl,coeff,acc_force,acc_force_E)
384 
385  USE output
386 
387 
388  REAL(KIND=dp),INTENT(IN) :: vcl(ntraj,n_dof), &
389  acc_force(ntraj,n_dof,nstates)
390  COMPLEX(KIND=QP),INTENT(IN) :: coeff(ntraj,nstates)
391  REAL(KIND=dp),INTENT(INOUT) :: acc_force_e(ntraj,n_dof,nstates)
392  INTEGER :: itraj,i_dof,istate
393  REAL(KIND=dp) :: tat(nstates),denom_ec(ntraj),nvec(ntraj,n_dof)
394  REAL(KIND=dp) :: ratio_ec(ntraj,n_dof)
395  REAL(KIND=dp) :: rho
396  REAL(KIND=dp), PARAMETER :: threshold = 0.005_dp
397  REAL(KIND=dp), PARAMETER :: r_threshold = 0.00001_dp
398 
399  !Initialize variables
400  acc_force_e=0.0_dp
401  tat=0.0_dp
402  denom_ec=0.0_dp
403  nvec=0.0_dp
404  ratio_ec=0.0_dp
405 
406  !Compute ratio=nvec/sum(nvec*Rdot)
407  DO itraj=1,ntraj
408  nvec(itraj,:)=mass(:)*vcl(itraj,:)
409  denom_ec(itraj)=dot_product(nvec(itraj,:),vcl(itraj,:))
410  ratio_ec(itraj,:)=nvec(itraj,:)/denom_ec(itraj)
411  ENDDO
412 
413  !Compute trajectory-averaged term with non-decohered trajectories
414  DO itraj=1,ntraj
415  IF(sum(acc_force(itraj,:,:)) /= 0.0_dp) THEN
416  DO istate=1,nstates
417  tat(istate)=tat(istate)+dot_product(acc_force(itraj,:,istate),vcl(itraj,:))+boenergy(itraj,istate)
418  ENDDO
419  ENDIF
420  ENDDO
421  tat(:)=tat(:)/dble(ntraj)
422 
423  !Computation of new accumulated force expression
424  DO istate=1,nstates
425  DO itraj=1,ntraj
426  rho=abs(coeff(itraj,istate))**2
427  IF(rho > threshold .AND. rho < 1.0_dp-threshold) THEN
428  !If trajectory is non-decohered
429  IF(abs(denom_ec(itraj))<r_threshold) THEN
430  !If denominator is too small revert back to original f definition
431  acc_force_e(itraj,:,istate)=acc_force(itraj,:,istate)
432  ELSE
433  !If denominator is acceptable proceed
434  DO i_dof=1,n_dof
435  acc_force_e(itraj,i_dof,istate)=(-boenergy(itraj,istate)+tat(istate))*ratio_ec(itraj,i_dof)
436  ENDDO
437  ENDIF
438  ELSE
439  !If trajectory decohered set acc force to zero
440  acc_force_e(itraj,:,istate)=0.0_dp
441  ENDIF
442  ENDDO
443  ENDDO
444 
445  END SUBROUTINE
