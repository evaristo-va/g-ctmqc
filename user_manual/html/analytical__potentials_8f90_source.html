61 
62  ALLOCATE(ebo(nstates,-1:1),u(nstates,nstates,-1:1), &
63  grad_hel(nstates,nstates,n_dof), &
64  factor(nstates))
65  ebo = 0.0_dp

67  grad_hel = 0.0_dp

69 
70  IF(model_potential=="NaI") &

87  ENDDO
88  CALL non_adiabatic_couplings(ebo(:,ix),grad_hel,u(:,:,ix),nac)
89  ENDIF

94  ENDDO

97  grad_bo = 0.0_dp

102  ENDDO
103  ENDIF

211  SUBROUTINE non_adiabatic_couplings(energy,gradients,eigenvectors,couplings)
212 
213  REAL(KIND=dp), INTENT(IN) :: energy(nstates), &
214  gradients(nstates,nstates,n_dof), &
215  eigenvectors(nstates,nstates)
216  REAL(KIND=dp), INTENT(INOUT) :: couplings(nstates,nstates,n_dof)
217  INTEGER :: i,j,k,l
218 
219  couplings=0.0_dp
220 
221  DO i=1,nstates
222  DO j=1,nstates
223  DO k=1,nstates
224  DO l=1,nstates
225  couplings(i,j,:)=couplings(i,j,:)+ &
226  eigenvectors(k,i)*gradients(k,l,:)* &
227  eigenvectors(l,j)
228  ENDDO
229  ENDDO
230  IF(i/=j) couplings(i,j,:)=couplings(i,j,:)/(energy(j)-energy(i))
231  ENDDO
232  ENDDO
233 
234  END SUBROUTINE non_adiabatic_couplings

253  grad_h(nstates,nstates)

255  REAL(KIND=dp), INTENT(IN) :: r
256  REAL(KIND=dp) :: acov,bcov,rhocov,ccov, &
257  aion,bion,rhoion,cion, &
258  alphamp,alphaxm,eth, &
259  a,rho
260 
261  acov = 3150._dp !eV
262  bcov = 2.647_dp !eV^1/12 * A
263  rhocov = 0.435_dp !A
264  ccov = 1000._dp !eV * A^6
265  aion = 2760._dp !eV
266  bion = 2.398_dp !eV^1/8 * A
267  rhoion = 0.3489_dp !A
268  cion = 11.3_dp !eV * A^6
269  eth = 2.075_dp !eV
270  alphamp = 0.408_dp !A^3
271  alphaxm = 6.431_dp !A^3
272  a = 17.08_dp !eV
273  rho = 1.239_dp !A
274 
275  acov = acov/au_to_ev
276  !Bcov =
277  rhocov = rhocov/au_to_ang
278  !Ccov =
279  aion = aion/au_to_ev
280  !Bion =
281  rhoion = rhoion/au_to_ang
282  !Cion =
283  eth = eth/au_to_ev
284  alphamp = alphamp/au_to_ang**3
285  alphaxm = alphaxm/au_to_ang**3
286  a = a/au_to_ev
287  rho = rho/au_to_ang
288 
289  h(1,1)=(acov+(1._dp/au_to_ev)*(bcov/(r*au_to_ang))**12)*exp(-r/rhocov) &
290  -(1._dp/au_to_ev)*ccov/(r*au_to_ang)**6
291  h(1,2)=a*exp(-r/rho)
292  h(2,1)=a*exp(-r/rho)
293  h(2,2)=(aion+(1._dp/au_to_ev)*(bion/(r*au_to_ang))**8)*exp(-r/rhoion) &
294  -1.0_dp/r-(alphamp+alphaxm)/2._dp/r**4 &
295  -(1._dp/au_to_ev)*cion/(r*au_to_ang)**6 &
296  -2._dp*alphamp*alphaxm/r**7+eth
297 
298  grad_h(1,1)=(-12._dp/r**13)*(1._dp/au_to_ev)*(bcov/au_to_ang)**12*exp(-r/rhocov) &
299  -(1._dp/rhocov)*(acov+(1._dp/au_to_ev)*(bcov/(r*au_to_ang))**12)*exp(-r/rhocov) &
300  -(-6._dp/r**7)*(1._dp/au_to_ev)*ccov/au_to_ang**6
301  grad_h(1,2)=-a/rho*exp(-r/rho)
302  grad_h(2,1)=-a/rho*exp(-r/rho)
303  grad_h(2,2)=(-8._dp/r**9)*(1._dp/au_to_ev)*(bion/au_to_ang)**8*exp(-r/rhoion) &
304  -(1._dp/rhoion)*(aion+(1._dp/au_to_ev)*(bion/(r*au_to_ang))**8)*exp(-r/rhoion) &
305  +1.0_dp/r**2-(-4._dp/r**5)*(alphamp+alphaxm)/2._dp &
306  -(-6._dp/r**7)*(1._dp/au_to_ev)*cion/au_to_ang**6 &
307  -(-7._dp/r**8)*2._dp*alphamp*alphaxm
308 
309  mass(1)=amu_to_au*(m_na*m_i)/(m_na+m_i)
310 
311  IF(PRESENT(r_crossing)) r_crossing=7.0_dp
312 
313  END SUBROUTINE nai_potential
