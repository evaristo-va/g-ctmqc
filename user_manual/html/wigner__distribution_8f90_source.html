64  print*, 'The variance on positions has to be positive'

84

143  FUNCTION gaussian_distribution(xi,nrand,var,x0,my_nrand) result(y)
144 
145  INTEGER, INTENT(IN) :: nrand,my_nrand
146  REAL(kind=dp),INTENT(IN) :: xi(nrand),var,x0
147  REAL(kind=dp) :: y_tmp(nrand),y(my_nrand)
148  INTEGER :: i,j
149 
150  j=1
151  DO i=1,my_nrand
152  y_tmp(i) = sqrt(-2.0_dp*log(xi(j+1)))*cos(2.0_dp*pi*xi(j))
153  j=j+2
154  ENDDO
155 
156  y_tmp = y_tmp * var + x0
157 
158  IF (nrand/=my_nrand) then
159  DO i=1,my_nrand
160  y(i) = y_tmp(i)
161  END DO
162  ELSE
163  y = y_tmp
164  END IF
165 
166  END FUNCTION gaussian_distribution
