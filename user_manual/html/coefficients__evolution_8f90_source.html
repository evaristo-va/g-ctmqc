48  SUBROUTINE rk4_coeff(v,coeff,k_li,trajlabel)
49 
50  INTEGER ,INTENT(IN) :: trajlabel
51  REAL(KIND=dp) ,INTENT(IN) :: v(n_dof),k_li(nstates,nstates)
52  COMPLEX(KIND=QP),INTENT(INOUT) :: coeff(nstates)
53  INTEGER :: i
54  COMPLEX(KIND=QP) :: k1,k2,k3,k4, &
55  variation(nstates), &
56  kfunction, &
57  my_coeff(nstates)
58  REAL(KIND=dp) :: normalization
59 
60  variation = cmplx(0.0_dp,0.0_dp,qp)
61  my_coeff = coeff
62 
63  statesloop: DO i=1,nstates
64  kfunction = cmplx(0.0_dp,0.0_dp,qp)
65  k1 = dt * cdot(i,kfunction,v,my_coeff,k_li,trajlabel)
66 
67  kfunction = 0.50_dp * k1
68  k2 = dt * cdot(i,kfunction,v,my_coeff,k_li,trajlabel)
69 
70  kfunction = 0.50_dp * (-1.0_dp + sqrt(2.0_dp)) * k1 + &
71  (1.0_dp - 0.5_dp * sqrt(2.0_dp)) * k2
72  k3 = dt * cdot(i,kfunction,v,my_coeff,k_li,trajlabel)
73 
74  kfunction = -0.50_dp * sqrt(2.0_dp) * k2 + &
75  (1.0_dp + 0.5_dp * sqrt(2.0_dp)) * k3
76  k4 = dt * cdot(i,kfunction,v,my_coeff,k_li,trajlabel)
77 
78  variation(i) = (k1 + (2.0_dp - sqrt(2.0_dp)) * k2 + &
79  (2.0_dp + sqrt(2.0_dp)) * k3 + k4)/6.0_dp
80  END DO statesloop
81 
82  coeff = my_coeff + variation
83 
84  normalization = 0.0_dp
85  DO i=1,nstates
86  normalization = normalization + (abs(coeff(i)))**2
87  END DO
88  coeff = coeff/sqrt(normalization)
89 
90  END SUBROUTINE rk4_coeff

116  FUNCTION cdot(state,kfunction,v,coeff,k_li,trajlabel)
117 
118  INTEGER ,INTENT(IN) :: state,trajlabel
119  REAL(kind=dp) ,INTENT(IN) :: v(n_dof),k_li(nstates,nstates)
120  COMPLEX(KIND=QP),INTENT(IN) :: coeff(nstates),kfunction
121  COMPLEX(KIND=QP) :: cdot,nonadiabatic_sum
122  COMPLEX(KIND=QP),ALLOCATABLE :: my_coeff(:)
123  INTEGER :: i
124  REAL(kind=dp) :: my_gap
125 
126  ALLOCATE(my_coeff(nstates))
127 
128  DO i=1,nstates
129  IF(i==state) THEN
130  my_coeff(i) = coeff(i) + kfunction
131  ELSE
132  my_coeff(i) = coeff(i)
133  END IF
134  END DO
135 
136  nonadiabatic_sum = cmplx(0.0_dp,0.0_dp,qp)
137 
138  DO i=1,nstates
139  my_gap=abs(boenergy(trajlabel,state)-boenergy(trajlabel,i))
140  IF(nrg_check .AND. my_gap>nrg_gap) coup_so(trajlabel,state,i)=0._dp
141  IF(i/=state) nonadiabatic_sum = nonadiabatic_sum + &
142  my_coeff(i) * (dot_product(coup(trajlabel,state,i,:),v(:)) + &
143  im_unit * coup_so(trajlabel,state,i))
144  END DO
145 
146  cdot = - im_unit * my_coeff(state) * boenergy(trajlabel,state) - nonadiabatic_sum
147 
148  IF(typ_cal=="CTMQC" .OR. type_deco=="CT") THEN
149  !WRITE(*,*) "I am here in coeff"
150  DO i=1,nstates
151  IF(i/=state) THEN
152  cdot = cdot + 0.250_dp * (k_li(i,state) - k_li(state,i)) * &
153  (abs(my_coeff(i)))**2 * my_coeff(state)
154  END IF
155  END DO
156  ENDIF
157 
158  DEALLOCATE(my_coeff)
159 
160  END FUNCTION cdot
