41  SUBROUTINE update_position(x,v)
42 
43  REAL(KIND=dp),INTENT(INOUT) :: x(n_dof)
44  REAL(KIND=dp),INTENT(IN) :: v(n_dof)
45  REAL(KIND=dp) :: my_x(n_dof)
46  INTEGER :: i_dof
47 
48  my_x = x + dt * v
49 
50  DO i_dof=1,n_dof
51  IF (periodic_in(i_dof)) THEN
52  IF (my_x(i_dof) < -period(i_dof) * 0.5_dp) my_x(i_dof) = my_x(i_dof) + period(i_dof)
53  IF (my_x(i_dof) >= period(i_dof) * 0.5_dp) my_x(i_dof) = my_x(i_dof) - period(i_dof)
54  END IF
55  END DO
56 
57  x = my_x
58 
59  END SUBROUTINE update_position

100  SUBROUTINE non_adiabatic_force(coeff,force,acc_force,k_li,trajlabel)
101 
102  INTEGER ,INTENT(IN) :: trajlabel
103  COMPLEX(KIND=QP),INTENT(IN) :: coeff(nstates)
104  REAL(KIND=dp) ,INTENT(IN) :: acc_force(n_dof,nstates),k_li(nstates,nstates)
105  REAL(KIND=dp) ,INTENT(INOUT) :: force(n_dof)
106  INTEGER :: i,j,i_dof,check
107  COMPLEX(KIND=QP),ALLOCATABLE :: my_rho(:,:)
108 
109  ALLOCATE(my_rho(nstates,nstates),stat=check)
110  IF(check/=0) print*,'error 1 my_rho in non_adiabatic_force'
111 
112  my_rho = cmplx(0.0_dp,0.0_dp,qp)
113  force = 0.0_dp
114 
115  DO i=1,nstates
116  DO j=1,nstates
117  my_rho(i,j) = conjg(coeff(i)) * coeff(j)
118  END DO
119  END DO
120 
121  !--------------------------------------------------------------------------!
122  ! DESCRIPTION: Ehrenfest-like force containing a mean force averaged over !
123  ! the populations of the electronic states and a non-adiabatic force !
124  ! depending on the non-adaiabtic couplings !
125  !--------------------------------------------------------------------------!
126  DO i_dof=1,n_dof
127  force(i_dof) = 0.0_dp
128  DO i=1,nstates
129  force(i_dof) = force(i_dof) + &
130  boforce(trajlabel,i,i_dof) * real(my_rho(i,i),kind=dp)
131  END DO
132  DO i=1,nstates
133  DO j=i+1,nstates
134  !IF(ABS(BOenergy(trajlabel,j)-BOenergy(trajlabel,i))<adia_nrg_gap) THEN
135  force(i_dof) = force(i_dof) - &
136  2.0_dp * real(my_rho(i,j),kind=dp) * &
137  (boenergy(trajlabel,j) - boenergy(trajlabel,i)) * &
138  coup(trajlabel,i,j,i_dof)
139  !ENDIF
140  ENDDO
141  ENDDO
142  ENDDO
143 
144  !--------------------------------------------------------------------------!
145  ! DESCRIPTION: quantum-momentum force only used in CT-MQC calculations !
146  !--------------------------------------------------------------------------!
147  IF(typ_cal=="CTMQC" .AND. qmom_force) THEN
148  DO i_dof=1,n_dof
149  DO i=1,nstates
150  DO j=1,nstates
151  force(i_dof) = force(i_dof) + 0.5_dp * k_li(i,j) * &
152  (acc_force(i_dof,j) - acc_force(i_dof,i)) * &
153  (real(my_rho(i,i),kind=dp) * real(my_rho(j,j),kind=dp))
154  END DO
155  END DO
156  END DO
157  ENDIF
158 
159  !--------------------------------------------------------------------------!
160  ! DESCRIPTION: SOC force only used in CT-MQC calculations !
161  !--------------------------------------------------------------------------!
162  IF(typ_cal=="CTMQC" .AND. spin_dia) THEN
163  DO i_dof=1,n_dof
164  DO i=1,nstates
165  DO j=1,nstates
166  force(i_dof)=force(i_dof)+ &
167  2.0_dp*acc_force(i_dof,i)*aimag(coup_so(trajlabel,i,j)*my_rho(i,j))
168  ENDDO
169  ENDDo
170  ENDDO
171  ENDIF
172 
173  DEALLOCATE(my_rho,stat=check)
174  IF(check/=0) print*,'error 2 my_rho in non_adiabatic_force'
175 
176  END SUBROUTINE non_adiabatic_force
